### 3. Задача 2. Вектор пользовательского типа Pair

#### 3.1. Класс Pair

Класс Pair содержит два поля: `int first` и `double second`. Для работы с контейнером и выполнения заданий перегружены операции сравнения (`<`, `>`), равенства (`==`, `!=`), сложения, деления на другой объект и на целое число, а также ввод/вывод.

```cpp
// Pair.h
#pragma once
#include <iostream>
using namespace std;

class Pair {
    int first;
    double second;
public:
    Pair() : first(0), second(0.0) {}
    Pair(int f, double s) : first(f), second(s) {}
    Pair(const Pair& p) : first(p.first), second(p.second) {}
    ~Pair() {}

    int get_first() const { return first; }
    double get_second() const { return second; }
    void set_first(int f) { first = f; }
    void set_second(double s) { second = s; }

    Pair& operator=(const Pair&);
    bool operator<(const Pair&) const;
    bool operator>(const Pair&) const;
    bool operator==(const Pair&) const;
    bool operator!=(const Pair&) const;
    Pair operator+(const Pair&) const;
    Pair operator/(const Pair&) const;
    Pair operator/(const int&) const;

    friend istream& operator>>(istream& in, Pair& p);
    friend ostream& operator<<(ostream& out, const Pair& p);
};
```

```cpp
// Pair.cpp
```

### 4. Задача 3. Параметризованный класс List (на основе `std::list`)

#### 4.1. Описание класса

Шаблонный класс `List` использует контейнер `std::list<T>` для хранения элементов. Все операции реализованы как методы класса.

```cpp
#include <iostream>
#include <list>
using namespace std;

template <class T>
class List {
    list<T> lst;
public:
    List() {}
    List(int n) {
        for (int i = 0; i < n; ++i) {
            T a;
            cin >> a;
            lst.push_back(a);
        }
    }
    List(const List<T>& l) : lst(l.lst) {}

    void Print() const {
        for (auto it = lst.begin(); it != lst.end(); ++it)
            cout << *it << " ";
        cout << endl;
    }

    T Min() const {
        auto it = lst.begin();
        T m = *it;
        for (++it; it != lst.end(); ++it)
            if (*it < m) m = *it;
        return m;
    }

    void AddMinAtPos(int pos) {
        T m = Min();
        if (pos < 0 || pos > (int)lst.size()) throw 1;
        auto it = lst.begin();
        advance(it, pos);
        lst.insert(it, m);
    }

    T Average() const {
        T sum = T();
        for (auto it = lst.begin(); it != lst.end(); ++it)
            sum = sum + *it;
        return sum / (int)lst.size();
    }

    void RemoveLessThanAverage() {
        T avg = Average();
        lst.remove_if([avg](const T& val) { return val < avg; });
    }

    T Max() const {
        auto it = lst.begin();
        T m = *it;
        for (++it; it != lst.end(); ++it)
            if (*it > m) m = *it;
        return m;
    }

    void DivideByMax() {
        T m = Max();
        for (auto it = lst.begin(); it != lst.end(); ++it)
            *it = *it / m;
    }
};
```

```cpp
#include <queue>
#include <vector>
#include "Pair.h"
using namespace std;

typedef queue<Pair> Que;
typedef vector<Pair> VecP;

Que make_queue(int n) {
    Que q;
    for (int i = 0; i < n; ++i) {
        Pair p;
        cin >> p;
        q.push(p);
    }
    return q;
}

VecP queue_to_vector(Que q) {
    VecP v;
    while (!q.empty()) {
        v.push_back(q.front());
        q.pop();
    }
    return v;
}

Que vector_to_queue(const VecP& v) {
    Que q;
    for (size_t i = 0; i < v.size(); ++i)
        q.push(v[i]);
    return q;
}

void print_queue(Que q) {
    while (!q.empty()) {
        cout << q.front() << " ";
        q.pop();
    }
    cout << endl;
}

Pair find_min_queue(Que q) {
    VecP v = queue_to_vector(q);
    Pair m = v[0];
    for (size_t i = 1; i < v.size(); ++i)
        if (v[i] < m) m = v[i];
    return m;
}

Que add_min_at_pos_queue(Que q, int pos) {
    Pair m = find_min_queue(q);
    VecP v = queue_to_vector(q);
    if (pos < 0 || pos > (int)v.size()) throw 1;
    v.insert(v.begin() + pos, m);
    return vector_to_queue(v);
}

Pair average_queue(Que q) {
    VecP v = queue_to_vector(q);
    Pair sum(0,0.0);
    for (size_t i = 0; i < v.size(); ++i)
        sum = sum + v[i];
    return sum / (int)v.size();
}

    Pair avg = average_queue(q);
    VecP v = queue_to_vector(q);
    VecP temp;
    for (size_t i = 0; i < v.size(); ++i)
        if (!(v[i] < avg)) temp.push_back(v[i]);
    return vector_to_queue(temp);
}

Pair find_max_queue(Que q) {
    VecP v = queue_to_vector(q);
    Pair m = v[0];
    for (size_t i = 1; i < v.size(); ++i)
        if (v[i] > m) m = v[i];
    return m;
}

Que divide_by_max_queue(Que q) {
    Pair m = find_max_queue(q);
    VecP v = queue_to_vector(q);
    for (size_t i = 0; i < v.size(); ++i)
```

### 6. Задача 5. Параметризованный класс QueueList (адаптер очереди)

#### 6.1. Описание класса

Шаблонный класс `QueueList` использует `std::queue<T>` в качестве хранилища. Для доступа к элементам внутри очереди применяется временный перенос в вектор.

```cpp
#include <queue>
#include <vector>
using namespace std;

template <class T>
class QueueList {
    queue<T> q;
public:
    QueueList() {}
    QueueList(int n) {
        for (int i = 0; i < n; ++i) {
            T a;
            cin >> a;
            q.push(a);
        }
    }
    QueueList(const QueueList<T>& other) : q(other.q) {}

    void Print() {
        queue<T> tmp = q;
        while (!tmp.empty()) {
            cout << tmp.front() << " ";
            tmp.pop();
        }
        cout << endl;
    }

    T Min() const {
        queue<T> tmp = q;
        T m = tmp.front();
        tmp.pop();
        while (!tmp.empty()) {
            if (tmp.front() < m) m = tmp.front();
            tmp.pop();
        }
        return m;
    }

    void AddMinAtPos(int pos) {
        T m = Min();
        vector<T> v;
        while (!q.empty()) {
            v.push_back(q.front());
            q.pop();
        }
        if (pos < 0 || pos > (int)v.size()) throw 1;
        v.insert(v.begin() + pos, m);
        for (size_t i = 0; i < v.size(); ++i)
            q.push(v[i]);
    }

    T Average() const {
        queue<T> tmp = q;
        T sum = T();
        int cnt = 0;
        while (!tmp.empty()) {
            sum = sum + tmp.front();
            tmp.pop();
            ++cnt;
        }
        return sum / cnt;
    }

    void RemoveLessThanAverage() {
        T avg = Average();
        vector<T> v;
        while (!q.empty()) {
            T val = q.front();
            q.pop();
            if (!(val < avg)) v.push_back(val);
        }
        for (size_t i = 0; i < v.size(); ++i)
            q.push(v[i]);
    }

    T Max() const {
        queue<T> tmp = q;
        T m = tmp.front();
        tmp.pop();
        while (!tmp.empty()) {
            if (tmp.front() > m) m = tmp.front();
            tmp.pop();
        }
        return m;
    }

    void DivideByMax() {
        T m = Max();
        vector<T> v;
        while (!q.empty()) {
            v.push_back(q.front() / m);
            q.pop();
        }
        for (size_t i = 0; i < v.size(); ++i)
            q.push(v[i]);
    }
};
```

### 7. Объяснение результатов работы программы

Все пять задач реализуют один и тот же алгоритм, но с разными контейнерами и типами элементов:

- В задаче 1 (vector<float>) операции выполняются прямо через индексацию.
- В задаче 2 (vector<Pair>) используется перегруженное сравнение и арифметика класса Pair.
- В задаче 3 (List<Pair>) применяются итераторы списка и методы `std::list`, такие как `remove_if`.
- В задаче 4 (queue<Pair>) из-за отсутствия прямого доступа к элементам используется вспомогательный вектор: очередь копируется в вектор, выполняются операции, результат снова помещается в очередь.
- В задаче 5 (QueueList<Pair>) те же действия инкапсулированы в методах шаблонного класса, работающего с очередью.

Все программы корректно находят минимальный элемент, вставляют его на заданную позицию (при допустимом значении позиции выбрасывается исключение), удаляют элементы меньше среднего арифметического и заменяют каждый элемент частным от деления на максимальный. Результаты подтверждают корректность перегрузки операций для Pair.

### 8. Ответы на контрольные вопросы

1. **Из каких частей состоит библиотека STL?**  
   STL состоит из набора контейнерных классов, обобщённых алгоритмов и итераторов.

2. **Какие типы контейнеров существуют в STL?**  
   Последовательные (vector, list, deque), ассоциативные (map, multimap, set, multiset, bitset), а также адаптеры (stack, queue, priority_queue).

3. **Что нужно сделать для использования контейнера STL в своей программе?**  
   Подключить соответствующий заголовочный файл (`#include <vector>`, `#include <list>` и т.д.) и использовать пространство имён `std` (или указывать `std::vector`).

4. **Что представляет собой итератор?**  
   Итератор – это объект, обобщающий понятие указателя; он ссылается на элементы контейнера и позволяет перемещаться по ним.

5. **Какие операции можно выполнять над итераторами?**  
   Разыменование (`*p`), присваивание, сравнение (`==`, `!=`), инкремент (`++p`, `p++`), а для итераторов произвольного доступа также декремент, сложение/вычитание с целым числом и сравнение `<`, `>`.

6. **Каким образом можно организовать цикл для перебора контейнера с использованием итератора?**  
   `for (Iterator i = container.begin(); i != container.end(); ++i) { ... }`.

7. **Какие типы итераторов существуют?**  
   Входные, выходные, прямые, двунаправленные, произвольного доступа.

8. **Перечислить операции и методы общие для всех контейнеров.**  
   `==`, `!=`, `=`, `clear()`, `insert()`, `erase()`, `size()`, `max_size()`, `empty()`, `begin()`, `end()`, реверсивные `rbegin()` и `rend()`.

9. **Какие операции являются эффективными для контейнера vector? Почему?**  
   Произвольный доступ (индексация), добавление и удаление в конце, потому что элементы хранятся в непрерывной памяти. Вставка/удаление в середине неэффективна из-за сдвига элементов.

10. **Какие операции являются эффективными для контейнера list? Почему?**  
    Вставка и удаление в любом месте, так как список двусвязный и не требует сдвига элементов. Произвольный доступ неэффективен.

11. **Какие операции являются эффективными для контейнера deque? Почему?**  
    Произвольный доступ, добавление/удаление как в начале, так и в конце; внутренняя реализация позволяет быстро работать с обоими концами без полного копирования.

12. **Перечислить методы, которые поддерживает последовательный контейнер vector.**  
    `push_back`, `pop_back`, `insert`, `erase`, `at`, `operator[]`, `swap`, `front`, `back`.

13. **Перечислить методы, которые поддерживает последовательный контейнер list.**  
    `push_back`, `pop_back`, `push_front`, `pop_front`, `insert`, `erase`, `swap`, `front`, `back`.

14. **Перечислить методы, которые поддерживает последовательный контейнер deque.**  
    `push_back`, `pop_back`, `push_front`, `pop_front`, `insert`, `erase`, `at`, `operator[]`, `swap`, `front`, `back`.

15. **Задан контейнер vector. Как удалить из него элементы со 2 по 5?**  
    `v.erase(v.begin() + 1, v.begin() + 5);` (индексы 2–5 соответствуют позициям 1–4 в итераторах).

16. **Задан контейнер vector. Как удалить из него последний элемент?**  
    `v.pop_back();` или `v.erase(v.end() - 1);`.

17. **Задан контейнер list. Как удалить из него элементы со 2 по 5?**  
    Использовать итераторы: `auto first = l.begin(); advance(first, 1); auto last = l.begin(); advance(last, 5); l.erase(first, last);`.

18. **Задан контейнер list. Как удалить из него последний элемент?**  
    `l.pop_back();` или `l.erase(prev(l.end()));`.

19. **Задан контейнер deque. Как удалить из него элементы со 2 по 5?**  
    Аналогично vector: `d.erase(d.begin() + 1, d.begin() + 5);`.

20. **Задан контейнер deque. Как удалить из него последний элемент?**  
    `d.pop_back();` или `d.erase(d.end() - 1);`.

21. **Написать функцию для печати последовательного контейнера с использованием итератора.**  
    ```cpp
    template<class T>
    void print(T& container) {
        auto it = container.begin();
        while (it != container.end())
            cout << *it++ << " ";
        cout << endl;
    }
    ```

22. **Что представляют собой адаптеры контейнеров?**  
    Это специализированные контейнеры, реализованные на основе базовых контейнеров (по умолчанию deque для stack, queue) и предоставляющие ограниченный интерфейс (например, только добавление сверху/снизу).

23. **Чем отличаются друг от друга объявления `stack<int> s` и `stack<int, list<int>> s`?**  
    Первый использует контейнер `deque` по умолчанию для хранения элементов, второй явно задаёт `list` в качестве нижележащего контейнера.

24. **Перечислить методы, которые поддерживает контейнер stack.**  
    `push()`, `pop()`, `top()`, `empty()`, `size()`.

25. **Перечислить методы, которые поддерживает контейнер queue.**  
    `push()`, `pop()`, `front()`, `back()`, `empty()`, `size()`.

26. **Чем отличаются друг от друга контейнеры queue и priority_queue?**  
    В queue элементы извлекаются в порядке FIFO, а в priority_queue извлекается наибольший (по умолчанию) элемент.

27. **Задан контейнер stack. Как удалить из него элемент с заданным номером?**  
    Так как прямого доступа нет, нужно переложить элементы во вспомогательный контейнер до нужного номера, удалить, затем вернуть обратно.

28. **Задан контейнер queue. Как удалить из него элемент с заданным номером?**  
    Аналогично: переложить в вектор или другой контейнер, удалить нужный элемент, затем восстановить очередь.

29. **Написать функцию для печати контейнера stack с использованием итератора.**  
    Итераторы в stack недоступны, поэтому печать осуществляется путём извлечения элементов:
    ```cpp
    template<class T>
    void print_stack(stack<T> s) {
        while (!s.empty()) {
            cout << s.top() << " ";
            s.pop();
        }
        cout << endl;
    }
    ```

30. **Написать функцию для печати контейнера queue с использованием итератора.**  
    Аналогично:
    ```cpp
    template<class T>
    void print_queue(queue<T> q) {
        while (!q.empty()) {
            cout << q.front() << " ";
            q.pop();
        }
        cout << endl;
    }
    ```
